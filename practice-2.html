Canvas

矩形

var c = document.getElementById("canvasID");
var ctx = c.getContext("2d");
ctx.beginPath();
ctx.strokeStyle = "black"; //設線條顏色
ctx.fillStyle = "red"; //設填滿顏色
ctx.rect(10,10,150,100); //設矩形範圍(x,y,width,height)
ctx.stroke(); //畫線條
ctx.fill(); //填滿顏色
//stroke跟fill的順序會影響畫出來的效果
ctx.closePath();

漸層

//線性漸層(color1-x,color1-y,color2-x,color2-y)
var grd = ctx.createLinearGradient(0,0,150,0);
//圓形漸層(color1-x,color1-y,color1-radius,color2-x,color2-y,color2-radius)
var grd = ctx.createRadialGradient(100,50,5,100,50,100);
grd.addColorStop(0, "#4d0099");
grd.addColorStop(1, "white");
ctx.fillStyle = grd;
ctx.fillRect(10,10,150,100);

直線

ctx.moveTo(0,0);
ctx.lineTo(150,100);
ctx.stroke();

圓

ctx.arc(50,50,40,0,2*Math.PI);
//(圓心x,圓心y,半徑,起始角,結束角,逆時針) 最後一項不寫, 預設是false, 即順時針, 填true為逆時針
//比較奇特的是0.5PI在下半圓,1.5PI在上半圓
ctx.stroke();

文字

ctx.font = "30px Arial";
ctx.fillText("Hello World",10,50);
ctx.strokeText("Hello World",10,50);

圖片

var img = document.getElementById("imgID");
ctx.drawImage(img,10,10);
